# –î–∞–≤–∞–π—Ç–µ –ø–æ—Å—Ç—Ä–æ–∏–º –Ω–æ–≤—É—é —Å—Ä–µ–¥–Ω—é—é –ª–∏–Ω–∏—é –Ω–∞ –æ—Å–Ω–æ–≤–Ω–æ–º –≥—Ä–∞—Ñ–∏–∫–µ –∏ —É—Å—Ç–∞–Ω–æ–≤–∏–º –º–∞—Ä–∫–µ—Ä—ã —Å–≤–µ—Ä—Ö—É, –ø–æ —Ü–µ–Ω—Ç—Ä—É –∏ —Å–Ω–∏–∑—É —Å–≤–µ—á–µ–π:
priceModel.prices["avg"] = priceModel.prices.low + (priceModel.prices.high - priceModel.prices.low) / 2
priceModel.prices["markersUpper"] = pd.Series(["‚Üì"] * len(priceModel.prices.high))
priceModel.prices["markersCenter"] = pd.Series(["√ó"] * len(priceModel.prices.avg))
priceModel.prices["markersLower"] = pd.Series(["‚Üë"] * len(priceModel.prices.low))
priceModel.RenderBokeh(
    fileName="index1.html",
    viewInBrowser=True,
    darkTheme=True,  # –£—Å—Ç–∞–Ω–æ–≤–∏—Ç–µ `False` –¥–ª—è –ø–µ—Ä–µ–∫–ª—é—á–µ–Ω–∏—è —Å–≤–µ—Ç–ª–æ–π —Ç–µ–º—ã.
    markers=priceModel.prices[["datetime", "markersUpper", "markersCenter", "markersLower"]],
    lines=[priceModel.prices[["datetime", "avg"]]],
    showStatOnChart=True,
    showControlsOnChart=True,
    inline=False,  # –†–∞—Å–∫–æ–º–º–µ–Ω—Ç–∏—Ä—É–π—Ç–µ, –µ—Å–ª–∏ —Å–∫—Ä–∏–ø—Ç –∑–∞–ø—É—Å–∫–∞–µ—Ç—Å—è –≤ Jupyter Notebook.
)
